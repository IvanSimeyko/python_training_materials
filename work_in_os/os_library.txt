import os
import os.path

os.getcwd() # узнать текущую директорию
os.listdir() # перечислим файлы в текуцей директории
os.path.exists('name.file or name.directory') # проверить существует ли файл или папка
os.path.isfile('name.file) # проверить является ли файлом
os.path.isdir('name.dir) # проверить является ли папкой
os.path.abspath('file.py') # узнать абсолютный путь файла
os.chdir('path') # сменить текущую директорию

ф-я os.walk() - рекурсивно проходит по всем папкам, подпапкам и так до конца. Ф-я
возврацает генератор и каждый раз возвращает кортеж из трех элементов - строковое
предстваление текущей директории которую он осматривает, список всех подпапок, который
есть в данной директории, список всех файлов в данной директории
for current_dir, dirs, files in os.work('.'):
    print(current_dir, dirs, files)
